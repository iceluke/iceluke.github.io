/* Command-line arguments, Basic output */

#include <iostream>

int
main (int argc, char **argv)
{
    std::cout << "argc = " << argc << std::endl;
    while (*argv!=NULL) {
        std::cout << *(argv++) << std::endl;
    }
    return 0;
}

// $ ./a.out 1 2 3 4
// argc = 4
// ./a.out
// 1
// 2
// 3
// 4

/* Creating simple classes, uniform initialization */

#include <iostream>
 
class Rectangle {
private:
    int _width;
    int _height;
public:
        Rectangle(int w, int h)
            : _width(w), _height(h)
        {

        }

        void print()
        {
            std::cout << "Rectangle: width=" << _width << ", height=" << _height << std::endl;
        }
};

int
main (int argc, char **argv)
{
    // uniform initialization (c++11)
    Rectangle r {5,6};
    r.print();
}

// $ ./a.out
// Rectangle: width=5, height=6
